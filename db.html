Potatoblock HTTP API 使用指南

欢迎来到 Potatoblock 的网络数据库开发文档页面！在这里，你将找到如何使用我们的 HTTP API 的详细说明。

API 概览

我们提供了多个 API 接口，以便用户进行数据存取、删除等操作。

1. 设置数据

请求方法：POST

请求路径：/set

请求头：

Content-Type: application/json
Authorization: Basic

请求体：

{
    "key": "YOUR_KEY",          // 替换为实际键
    "value": {
        "name": "Alice",        // 示例字典（或列表）内容
        "age": 30
    }                           // 或者
    "value": [1, 2, 3, 4, 5]   // 示例列表
}

响应示例：

HTTP/1.1 201 Created
Content-Type: application/json

{
    "message": "Data set successfully"
}

Lua 示例代码

以下是如何在 Lua 中调用 PotatoDB API 的示例代码：

获取数据

-- 获取数据
local function get_data(key)
    local url = "http://:@/get?key=" .. key
    local response, error_message = coromega:http_request("GET", url)

    if error_message then
        print("Error getting data: ", error_message)
        return nil
    else
        if response.status_code == 200 then
            local result = json.decode(response.body)
            return result[key]  -- 返回获取的值
        else
            print("Error response: ", response.body)
            return nil
        end
    end
end

设置数据

-- 设置数据
local function set_data(key, value)
    local url = "http://:@/set"
    local payload = {
        key = key,
        value = value
    }
    local response, error_message = coromega:http_request("POST", url, {
        body = json.encode(payload)
    })

    if error_message then
        print("Error adding data: ", error_message)
        return nil
    else
        if response.status_code == 201 then
            return true
        else
            print("Error response: ", response.body)
            return false
        end
    end
end

删除数据

-- 删除数据
local function delete_data(key)
    local url = "http://:@/delete"
    local payload = {
        key = key
    }
    local response, error_message = coromega:http_request("POST", url, {
        body = json.encode(payload)
    })

    if error_message then
        print("Error deleting data: ", error_message)
        return nil
    else
        if response.status_code == 202 then
            return true
        else
            print("Error response: ", response.body)
            return false
        end
    end
end

执行 SQL 语句

-- 执行 SQL 语句
local function execute_sql_query(query)
    local url = "http://:@/sql"
    local payload = {
        query = query
    }
    local response, error_message = coromega:http_request("POST", url, {
        body = json.encode(payload)
    })

    if error_message then
        print("Error executing SQL query: ", error_message)
        return nil
    else
        if response.status_code == 200 then
            local result = json.decode(response.body)
            return result.result  -- 返回结果
        else
            print("Error response: ", response.body)
            return nil
        end
    end
end
